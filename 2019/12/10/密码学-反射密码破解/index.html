class Node:
	ch = ''
	cnt = 0
	def __init__(self,ch,cnt):
		self.ch = ch
		self.cnt = cnt
	def __str__(self):
		return self.ch+":\t"+str(self.cnt)
	def __lt__(self,other):
		return self.cnt > other.cnt


def gcd(a,b):
	if(a%b == 0):
		return b
	return gcd(b,a%b)
def AffineEncrypt(m,k1,k2):
	k2%=26
	if(gcd(k1,26)!=1):#判断k1是否合法
		print("k1和26不是互素的整数，请重新选择k1")
		return
	l=len(m)
	rep=""
	for i in range(l):
		if(m[i].isupper()):#判断是大写字母还是小写字母
			rep += chr((k1*(ord(m[i])-ord('A'))+k2)%26+ord('A'))
		elif(m[i].islower()):
			rep += chr((k1*(ord(m[i])-ord('a'))+k2)%26+ord('a'))
		else:
			rep+=m[i]
	return rep

def Euclidean(kr,m):
	if(gcd(kr,m)!=1):
		return None
	i1,i2,i3=1,0,kr
	j1,j2,j3=0,1,m
	while(j3!=0):
		q =i3//j3
		j1,j2,j3,i1,i2,i3=(i1-q*j1),(i2-q*j2),(i3-q*j3),j1,j2,j3
	return i1%m
def AffineDecrypt(c,k1,k2):#ci=(k1·mi+k2)mod26
	k2 %= 26
	if(gcd(k1,26)!=1):#判断k1是否合法
		print("k1和26不是互素的整数，请重新选择k1")
		return
	l=len(c)
	kr=Euclidean(k1,26)
	rep=""
	for i in range(l):
		if(c[i].isupper()):#判断是大写字母还是小写字母
			rep += chr((kr*(ord(c[i])-ord('A')+26-k2))%26+ord('A'))
		elif(c[i].islower()):
			rep += chr((kr*(ord(c[i])-ord('a')+26-k2))%26+ord('a'))
		else:
			rep+=c[i]
	return rep
def StatisticalAffineDecrypt(c):
	l = len(c)
	ciperCopy = c.lower()
	alphaNum = []
	# Initialize
	for i in range(26):
		t = Node(chr(ord('a')+i),0)
		alphaNum.append(t)
	# Count
	for i in range(l):
		if ciperCopy[i].isalpha():
			alphaNum[ord(ciperCopy[i])-ord('a')].cnt += 1
	# Sort
	alphaNum.sort()
