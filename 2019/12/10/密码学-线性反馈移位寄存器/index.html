最近在上密码学课设，遇到了线性反馈移位寄存器，决定用python来实现

代码如下

def XOR(ai,t):
    sum=0
    for i in range(len(ai)):
        if(ai[i]==1):
            sum+=ai[i]*t[i]
    return sum%2
def LFSR(ai,t):
    #x^22+x+1
    
    t.append(XOR(ai,t))
    t.pop(0)
  
    return t

    t=list(posture)#深拷贝，浅拷贝实现不了

        strlfsr+=str(posture[0])
        print(LFSR(basefunction,posture),'循环次数为',sum)
        sum=sum+1

        if(t==posture):
            
            if(sum==pow(2,len(basefunction))):
